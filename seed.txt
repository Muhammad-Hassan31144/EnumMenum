pythonasync def enumerate_subdomains(domain):
    results = await asyncio.gather(
        crtsh_search(domain),
        subfinder_scan(domain),
        amass_enum(domain),
        dnsdumpster_search(domain),
        virustotal_search(domain)
    )
    
    # Merge and deduplicate
    all_subs = set()
    source_map = {}
    
    for source, subdomains in results:
        for sub in subdomains:
            all_subs.add(sub)
            source_map.setdefault(sub, []).append(source)
    
    return list(all_subs), source_map

3. Takeover Detection
pythonTAKEOVER_SIGNATURES = {
    'github': {
        'cname': ['github.io'],
        'response': 'There isn\'t a GitHub Pages site here'
    },
    's3': {
        'cname': ['s3.amazonaws.com'],
        'response': 'NoSuchBucket'
    },
    'heroku': {
        'cname': ['herokuapp.com'],
        'response': 'No such app'
    }
}

def check_takeover(subdomain, cname, http_response):
    for service, signature in TAKEOVER_SIGNATURES.items():
        if any(pattern in cname for pattern in signature['cname']):
            if signature['response'] in http_response:
                return {
                    'vulnerable': True,
                    'service': service,
                    'severity': 'high'
                }
    return {'vulnerable': False}
4. Change Tracking (Diff Tool)
javascriptconst DiffView = ({ oldScan, newScan }) => {
  const added = newScan.filter(s => 
    !oldScan.find(o => o.subdomain === s.subdomain)
  );
  
  const removed = oldScan.filter(s => 
    !newScan.find(n => n.subdomain === s.subdomain)
  );
  
  const changed = newScan.filter(s => {
    const old = oldScan.find(o => o.subdomain === s.subdomain);
    return old && (old.ip !== s.ip || old.httpStatus !== s.httpStatus);
  });
  
  return (
    <div>
      <h3>Added ({added.length})</h3>
      {added.map(s => <div className="text-green-500">+ {s.subdomain}</div>)}
      
      <h3>Removed ({removed.length})</h3>
      {removed.map(s => <div className="text-red-500">- {s.subdomain}</div>)}
      
      <h3>Changed ({changed.length})</h3>
      {changed.map(s => <div className="text-yellow-500">~ {s.subdomain}</div>)}
    </div>
  );
};

pythonfrom celery import Celery
import subprocess

app = Celery('subdomain_scanner')

@app.task
def scan_domain(domain, scan_id):
    # Update status
    update_scan_status(scan_id, 'running')
    
    # Run tools
    subdomains = []
    
    # crt.sh
    crtsh_results = requests.get(f'https://crt.sh/?q=%.{domain}&output=json')
    subdomains.extend(parse_crtsh(crtsh_results.json()))
    
    # subfinder
    result = subprocess.run(['subfinder', '-d', domain, '-silent'], 
                           capture_output=True, text=True)
    subdomains.extend(result.stdout.splitlines())
    
    # amass (passive)
    result = subprocess.run(['amass', 'enum', '-passive', '-d', domain],
                           capture_output=True, text=True)
    subdomains.extend(result.stdout.splitlines())
    
    # Deduplicate
    unique_subs = list(set(subdomains))
    
    # Resolve IPs + HTTP probe
    for sub in unique_subs:
        ip = resolve_dns(sub)
        http_status, title, tech = probe_http(sub)
        takeover = check_takeover(sub)
        
        save_subdomain(scan_id, sub, ip, http_status, title, tech, takeover)
    
    update_scan_status(scan_id, 'completed')

    });
    const { scan_id } = await response.json();
    setScanId(scan_id);
    
    // Poll for updates
    const interval = setInterval(async () => {
      const status = await fetch(`/api/scan/${scan_id}/status`);
      const data = await status.json();
      
      if (data.status === 'completed') {
        clearInterval(interval);
        setScanning(false);
        loadResults(scan_id);
      }
    }, 2000);
  };
  
  return (

      {scanning ? 'Scanning...' : 'Start Scan'}
